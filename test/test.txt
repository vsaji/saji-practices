module.exports = function override(config, env) {
  // Check if resolve exists, and if not, create it
  if (!config.resolve) {
    config.resolve = {};
  }

  // Check if resolve.fallback exists, and if not, create it
  if (!config.resolve.fallback) {
    config.resolve.fallback = {};
  }

  // Add "https" fallback to https-browserify
  config.resolve.fallback['https'] = require.resolve('https-browserify');

  return config;
